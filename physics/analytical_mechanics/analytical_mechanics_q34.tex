\begin{question}
    Determine a intensidade do momento angular de uma esfera homogênea de massa \measure{2.20}{kg} e raio \measure{5}{cm} que gira com velocidade angular de \measure{3}{rad/s}.

    \begin{answer}
      \measure{6.60e-3}{J.s} ou \measure{6.60e-3}{kg.m^2/s}
    \end{answer}
    
    \begin{solution}
      $L = I\omega$, onde $I$ é o momento de inércia da esfera, dado por $I = \frac{2}{5}mr^2$ e $\omega = \measure{3}{rad/s}$ é a velocidade angular.
      $m = \measure{2.20}{kg}$ é a massa da esfera e $r = \measure{0.0500}{m}$, seu raio.
      Consequentemente, $L = \measure{6.60e-3}{J.s}$.
    \end{solution}
\end{question}
