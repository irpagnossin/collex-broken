\begin{question}
    Considere novamente o cenário da conta presa a um arame senoidal $z = \sin(x)$.
    Determine a expressão da energia potencial gravitacional.
    \begin{enumerate}
      \item $U = mg\sin x$ \rightanswer
      \item $U = mgz$
      \item $U = mgh$
      \item $U = -mg\sin x$
      \item $U = -mgz$
    \end{enumerate}

    \begin{solution}
      Começamos com $U = mgz$.
      Mas $z = \sin(x)$, então $U = mg\sin(x)$.
    \end{solution}
\end{question}
