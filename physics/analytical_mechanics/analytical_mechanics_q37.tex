\begin{question}
    Determine a intensidade do torque associado à força de \measure{500}{N} aplicada perpendicularmente à extremidade do corpo de uma chave de boca de \measure{15}{cm}.

    \begin{answer}
      \measure{75.0}{N.m}. Atenção: é \emph{incorreto} usar \unit{J} (joule) no lugar de \unit{N.m}.
    \end{answer}
    
    \begin{solution}
      $\vec\tau = \vec r \times \vec F$.
      Como $\vec r \perp \vec F$, $|\vec\tau| \equiv \tau = rF = \measure{75.0}{N.m}$.
      Atenção: é \emph{incorreto} usar \unit{J} (joule) no lugar de \unit{N.m}.
    \end{solution}
\end{question}
