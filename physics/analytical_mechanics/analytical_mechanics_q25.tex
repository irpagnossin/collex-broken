\begin{question}
    Determine o momento de inércia de um objeto puntual de massa \measure{1.00}{kg} orbitando um eixo, a \measure{2.00}{m} dele.

    \begin{answer}
      \measure{4.00}{kg.m^2}
    \end{answer}

    \begin{solution}
      O momento de inércia de uma partícula de massa $m$ com relação a um eixo à distância $r$ é $I = mr^2$.
      Usando $m = \measure{1.00}{kg}$ e $r = \measure{2.00}{m}$ nessa expressão, obtemos $I = \measure{4.00}{kg.m^2}$.
    \end{solution}
\end{question}
