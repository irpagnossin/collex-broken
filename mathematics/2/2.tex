	Desenhe o vetores
	\begin{inlineenum}
		\inlineitem $u + v$,
		\inlineitem $u - v$ e
		\inlineitem $\frac{3}{2}\vec u$
	\end{inlineenum}

	\begin{center}
		\input{mathematics/2/figures/u_v.tex}
	\end{center}

	\begin{answer}

			\item \inlinebox{\input{\mypath/figures/u_v_item_a_head_to_tail}} ou \inlinebox{\input{\mypath/figures/u_v_item_a_parallelogram}}

			\item \inlinebox{\input{\mypath/figures/u_v_item_c}}
		\end{enumerate}
	\end{answer}

	\begin{solution}
		\begin{enumerate}
			\item
			Pela definição, \inlinebox{\input{\mypath/figures/u_v_item_a_head_to_tail}}, ou pela regra do paralelogramo: \inlinebox{\input{\mypath/figures/u_v_item_a_parallelogram}}

			\item

		\end{enumerate}
	\end{solution}
